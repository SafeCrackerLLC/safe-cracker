395void recordLevelStability(int potDiff) {
396 if (gameScreen != GAME_SCREEN_PLAYING || isLevelComplete() || isLevelOutOfTime()) {
397 return;
398 }
399
400 float targetFill = getTargetFillAmount();
401 int targetIndex = currentTargetIndex;
402 stabilityErrorSum += constrain(abs(barFill - targetFill) / 2.0, 0.0, 1.0);
403 stabilitySampleCount++;
404
405 if (targetIndex >= 0 && targetIndex < MAX_LEVEL_TARGETS) {
406 targetStabilityErrorSum[targetIndex] += constrain(abs(barFill - targetFill) / 2.0, 0.0, 1.0);
407 targetStabilitySampleCount[targetIndex]++;
408 }
409
410 if (
411 (targetFill > 0 && barFill > targetFill + TARGET_OVERSHOOT_ALLOWANCE) ||
412 (targetFill < 0 && barFill < targetFill - TARGET_OVERSHOOT_ALLOWANCE)
413 ) {
414 stabilityOvershootCount++;
415
416 if (targetIndex >= 0 && targetIndex < MAX_LEVEL_TARGETS) {
417 targetStabilityOvershootCount[targetIndex]++;
418 }
419 }
420
421 int direction = 0;
422 if (potDiff > POT_DEADZONE) {
423 direction = 1;
424 } else if (potDiff < -POT_DEADZONE) {
425 direction = -1;
426 }
427
428 if (direction != 0) {
429 if (lastPotDirection != 0 && direction != lastPotDirection) {
430 stabilityDirectionChanges++;
431 }
432
433 lastPotDirection = direction;
434
435 if (targetIndex >= 0 && targetIndex < MAX_LEVEL_TARGETS) {
436 if (targetLastPotDirection[targetIndex] != 0 && direction != targetLastPotDirection[targetIndex]) {
437 targetStabilityDirectionChanges[targetIndex]++;
438 }
439
440 targetLastPotDirection[targetIndex] = direction;
441 }
442 }

448 }

456 }
457
458 if (timeNow - lastReplayEventTime < REPLAY_SAMPLE_INTERVAL_MS) {

460 }

463 replayEventTimeMs[replayEventCount] = timeMs;
464 replayEventFill[replayEventCount] = barFill;
465 replayEventTargetIndex[replayEventCount] = currentTargetIndex;
466 replayEventType[replayEventCount] = eventType;

472int calculateLevelStability() {
473 if (stabilitySampleCount <= 0) {
474 return 100;
475 }
476
477 float averageError = stabilityErrorSum / stabilitySampleCount;
478 float overshootRate = (float)stabilityOvershootCount / stabilitySampleCount;
479 float directionRate = (float)stabilityDirectionChanges / stabilitySampleCount;
480 float score = 100.0 - averageError * 55.0 - overshootRate * 30.0 - directionRate * 15.0;
481
482 return constrain((int)(score + 0.5), 0, 100);
483}

485int calculateTargetStability(int targetIndex) {
486 if (targetIndex < 0 || targetIndex >= MAX_LEVEL_TARGETS || targetStabilitySampleCount[targetIndex] <= 0) {
487 return 100;
488 }
489
490 float averageError = targetStabilityErrorSum[targetIndex] / targetStabilitySampleCount[targetIndex];
491 float overshootRate = (float)targetStabilityOvershootCount[targetIndex] / targetStabilitySampleCount[targetIndex];
492 float directionRate = (float)targetStabilityDirectionChanges[targetIndex] / targetStabilitySampleCount[targetIndex];
493 float score = 100.0 - averageError * 55.0 - overshootRate * 30.0 - directionRate * 15.0;
494
495 return constrain((int)(score + 0.5), 0, 100);
496}

537void updateVaultRotationFromPot() {
538 int currentValue = analogRead(POT_PIN);
539 int diff = currentValue - lastValue;
540
541 updateTurnBarFromPot(currentValue);
542 recordLevelStability(diff);
543 recordReplayEvent(REPLAY_EVENT_MOVE, false);
544 updateTargetProgress();
545
546 if (gameScreen != GAME_SCREEN_PLAYING) {
547 return;
548 }
549
550 if (abs(diff) <= POT_DEADZONE) {
551 return;
552 }
553
554 vaultAngle += diff * ROTATION_SPEED;
555
556 while (vaultAngle >= FULL_ROTATION) {
557 vaultAngle -= FULL_ROTATION;
558 }
559
560 while (vaultAngle < 0) {
561 vaultAngle += FULL_ROTATION;
562 }
563
564 if (diff > 0) {
565 clickRight();
566 } else {
567 clickLeft();
568 }
569

571 resetSleepTimer();
