97/**
98 * @brief Interpolira replay stanje za zadani trenutak.
99 * @param {number} timeMs Vrijeme u milisekundama od pocetka pokusaja.
100 * @returns {object} Interpolirano stanje replaya.
101 */
102function interpolateEvent(timeMs) {
103 if (!replay.length) {
104 return { t: 0, fill: 0, target_index: 0, event: "empty" };
105 }
106
107 let previous = replay[0];
108 let next = replay[replay.length - 1];
109
110 for (let i = 1; i < replay.length; i += 1) {
111 if (replay[i].t >= timeMs) {
112 next = replay[i];
113 break;
114 }
115 previous = replay[i];
116 }
117
118 if (next.t === previous.t) {
119 return previous;
120 }
121
122 const progress = Math.max(0, Math.min(1, (timeMs - previous.t) / (next.t - previous.t)));
123 return {
124 t: timeMs,
125 fill: Number(previous.fill) + (Number(next.fill) - Number(previous.fill)) * progress,
126 target_index: previous.target_index,
127 event: previous.event,
128 };
129}

131/**
132 * @brief Renderira replay marker, aktivni target i tekstualni readout.
133 * @param {number} timeMs Vrijeme u milisekundama od pocetka pokusaja.
134 */
135function renderReplay(timeMs) {
136 if (!replayStage || !elements.marker) return;
137
138 const state = interpolateEvent(timeMs);
139 const activeIndex = Number(state.target_index) || 0;
140 const activeTargetFill = targets[activeIndex] ?? 0;
141
142 elements.marker.style.left = fillToPercent(state.fill);
143 elements.activeTarget.style.left = fillToPercent(activeTargetFill);
144 elements.readout.textContent = `fill ${Number(state.fill).toFixed(2)} | target #${activeIndex + 1} | ${eventLabel(state.event)}`;
145
146 elements.targets.forEach((targetNode) => {
147 targetNode.classList.toggle("is-active", Number(targetNode.dataset.index) === activeIndex);
148 });
149}

151/**
152 * @brief Postavlja trenutno vrijeme replaya i sinkronizira scrub kontrolu.
153 * @param {number} timeMs Novo vrijeme replaya u milisekundama.
154 */
155function setTime(timeMs) {
156 const bounded = Math.max(0, Math.min(durationMs, timeMs));
157 pausedAt = bounded;
158
159 if (scrub) {
160 scrub.value = durationMs ? Math.round((bounded / durationMs) * 1000) : 0;
161 }
162
163 if (timeLabel) {
164 timeLabel.value = `${(bounded / 1000).toFixed(1)}s`;
165 }
166
167 renderReplay(bounded);
168}

170/**
171 * @brief Animation-frame callback za automatsku reprodukciju replaya.
172 * @param {DOMHighResTimeStamp} timestamp Vrijeme koje salje requestAnimationFrame.
173 */
174function tick(timestamp) {
175 if (!startedAt) {
176 startedAt = timestamp - pausedAt;
177 }
178
179 const elapsed = timestamp - startedAt;
180 setTime(elapsed);
181
182 if (elapsed >= durationMs) {
183 playing = false;
184 startedAt = 0;
185 if (toggle) toggle.textContent = "Play";
186 return;
187 }
188
189 animationFrame = requestAnimationFrame(tick);
190}
191
192if (toggle) {
193 toggle.addEventListener("click", () => {
194 if (!durationMs) return;
195
196 playing = !playing;
197 toggle.textContent = playing ? "Pause" : "Play";
198
199 if (playing) {
200 startedAt = 0;
201 animationFrame = requestAnimationFrame(tick);
202 } else {
203 cancelAnimationFrame(animationFrame);
204 }
205 });
206}
207
208if (scrub) {
209 scrub.addEventListener("input", () => {
210 cancelAnimationFrame(animationFrame);
211 playing = false;
212 if (toggle) toggle.textContent = "Play";
213 setTime(durationMs * (Number(scrub.value) / 1000));
214 });
215}
